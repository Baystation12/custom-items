# {
# ckey: zuhayr
# character_name: Jane Doe
# item_path: /obj/item/toy/plushie
# item_name: ugly plush toy
# item_icon: flagmask     
# item_desc: It's truly hideous.
# req_titles: Assistant, Security Officer
# req_access: 1
# }
#
# {
# ckey: zuhayr
# character_name: Jane Doe
